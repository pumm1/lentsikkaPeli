8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.Random;
11  import javax.swing.SwingUtilities;
12  import sakureme.logiikka.Hahmo;
13  import sakureme.logiikka.Pala;
14  import sakureme.logiikka.Pelaaja;
15  import sakureme.logiikka.Puu;
16  
17  /**
18   * Peli-luokalla tehdään Main-luokasta siistimpi, alustamalla tarvittavat oliot
19   * täällä käynnistystä varten
20   */
21  public class Peli {
22  
23      private Kayttoliittyma kayttis;
24      private Random rand = new Random();
25  
26      public Peli() {
27  
28          kayttis = new Kayttoliittyma();
29      }
30  
31      //luodaan hahmot ja asetetaan ne paikalleen
32      public void alustaHahmot() {
33          int x = rand.nextInt(300) + 200;
34          int y;
35          Puu puu1 = new Puu(400, 250, 100, 350);   //olevinaan puu/tolppa
36          Puu puu2 = new Puu(750, 200, 100, 350);
37          Puu puu3 = new Puu(1100, 230, 100, 350);
38          puu1.lataaKuva();
39          puu2.lataaKuva();
40          puu3.lataaKuva();
41  
42  //        List<Pala> palat = new ArrayList<>();
43          List<Hahmo> hahmot = new ArrayList<>();
44  
45          x = rand.nextInt(500) + 200;
46          y = rand.nextInt(250) + 100;
47  
48          //luodaan pelattava hahmo
49          List<Pala> hahmoPalat = new ArrayList<>();
50          hahmoPalat.add(new Pala(50, 50, 20, 20));
51          hahmoPalat.add(new Pala(50, 70, 100, 20));
52          hahmoPalat.add(new Pala(120, 60, 20, 20));
53  
54          Pelaaja pelaaja = new Pelaaja(hahmoPalat);
55          pelaaja.setPutoavuus(true);
56          pelaaja.lataaKuva();
57  
58          hahmot.add(puu1);
59          hahmot.add(puu2);
60          hahmot.add(pelaaja);
61          hahmot.add(puu3);
62  
63          kayttis.lisaaHahmot(hahmot);
64          kayttis.setPelattavaHahmo(pelaaja);
65      }
66  
67      public void aloita() {
68          SwingUtilities.invokeLater(kayttis);
69      }
70  
71  }
