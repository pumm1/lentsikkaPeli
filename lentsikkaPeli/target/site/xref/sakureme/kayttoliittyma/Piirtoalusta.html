6   import java.util.*;
7   import java.awt.Color;
8   import java.awt.Container;
9   import java.awt.Graphics;
10  import java.awt.event.ActionEvent;
11  import java.awt.event.ActionListener;
12  import javax.swing.JPanel;
13  import javax.swing.Timer;
14  import sakureme.logiikka.Hahmo;
15  import sakureme.logiikka.Pelaaja;
16  import sakureme.logiikka.Pilvi;
17  
18  public class Piirtoalusta extends JPanel implements ActionListener {
19  
20      private List<Hahmo> hahmot;
21      private Timer timer;
22      private Random rand = new Random();
23      private Pelaaja pelaaja;
24  
25      public Piirtoalusta(List<Hahmo> a) {
26          super.setBackground(Color.white);
27          hahmot = a;
28          timer = new Timer(35, this);
29          timer.start();
30      }
31  
32      @Override
33      protected void paintComponent(Graphics g) {
34  
35          super.paintComponent(g);
36          for (Hahmo i : hahmot) {
37              i.piirraPalat(g);
38          }
39  
40      }
41  
42      public void setHahmo(Pelaaja i) {
43          pelaaja = i;
44      }
45  
46      @Override
47      public void actionPerformed(ActionEvent e) {
48  
49          if (e.getSource() == timer) {
50              if (!pelaaja.peliAlueenSisalla()) { //hävisit pelin
51                  pelaaja.kuolee();
52                  repaint();
53                  timer.stop();
54              }
55              for (Hahmo i : hahmot) {
56                  i.putoa();
57                  if (i.getPelattavuus() == false) {
58                      i.liiku();
59                      if (i instanceof Pilvi) {
60                          if (((Pilvi) i).getX() <= -100) {
61                              i.siirraUuteenPaikkaan();
62                          }
63                      } else if (i.getPalat().get(0).getX() <= -300) {
64                          i.siirraUuteenPaikkaan();
65                      }
66                  }
67                  if (!i.equals(pelaaja)) {
68                      if (pelaaja.osuu(i)) {  //hävisit pelin
69                          pelaaja.kuolee();
70                          repaint();
71                          timer.stop();
72                      }
73                  }
74              }
75          }
76          repaint();
77      }
78  }
