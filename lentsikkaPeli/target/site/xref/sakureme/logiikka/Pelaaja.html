41      /**
42       * konstruktori saa parametreina palat, jotka muodostavat lentokoneen
43       *
44       * @param i lista pelaajan paloista
45       */
46      public Pelaaja(List<Pala> i) {
47          super(i);
48          ladattu = false;
49          alku = System.currentTimeMillis();
50          loput = new ArrayList<>();
51          String g0 = "Hävisit pelin";
52          String g1 = "Nyt tuli noutaja!";
53          String g2 = "Niinpä tietysti.";
54          String g3 = "GAME OVER";
55          String g4 = "Sattuko pahasti?";
56          loput.add(g0);
57          loput.add(g1);
58          loput.add(g2);
59          loput.add(g3);
60          loput.add(g4);
61      }

63      /**
64       * ei mahdollisteta sitä, että pelaaja huijaa pysymällä palikoiden
65       * ulottumattomissa!
66       *
67       * @return
68       */
69      public boolean peliAlueenSisalla() {
70          if (this.getPalat().get(0).getY() < -10) {
71              return false;
72          } else if (this.getPalat().get(0).getY() > 420) {
73              return false;
74          }
75          return true;
76      }

78      /**
79       * ladataan pelaajan .png-kuva
80       */
81      public void lataaKuva() throws IOException {
82          //"D:\\Projekstit\\lentsikkaPeli\\lentsikkaPeli\\src\\lentsikka.png" <- talteen varalle
83          //seuraavan avulla kaikilla koodin omaavilla pitäisi kuvien näkyä
84          InputStream is = getClass().getClassLoader().getResourceAsStream("lentsikka.png");
85  //        System.out.println(file.getAbsolutePath());
86          BufferedImage im = ImageIO.read(is);
87          kuva = (BufferedImage) im;
88  
89          //varmennetaan vielä, että jos jokin menee pieleen, niin käytetään palikkagrafiikkaa
90          if (kuva != null) {
91              ladattu = true;
92          }
93      }

103     public void palautaEloon() {
104         elossa = true;
105         alku = System.currentTimeMillis();
106         loppu = alku;
107     }

109     public void kuolee() {  // rip :(
110         elossa = false;
111         loppu = System.currentTimeMillis();
112         long temp = loppu - alku;
113         aika = (int) temp;
114         msec = aika % 1000;
115         sec = (aika - msec) / 1000;
116     }
117 
118     @Override
119     public void piirraPalat(Graphics g) {
120         String time;
121         if (elossa) {
122             loppu = System.currentTimeMillis();
123         }
124         long temp = loppu - alku;
125         aika = (int) temp;
126         msec = aika % 1000;
127         sec = (aika - msec) / 1000;
128 
129         time = "Aika: " + sec + "," + msec + "s";
130 
131         g.setFont(new Font("Arial", Font.PLAIN, 20));
132         g.setColor(Color.black);
133         g.drawString(time, 30, 420);
134         if (ladattu) {
135             int x = this.getPalat().get(0).getX();
136             int y = this.getPalat().get(0).getY();
137             g.drawImage(kuva, x, y, null);
138         } else {
139             for (Pala i : palat) {
140                 i.piirra(g);
141             }
142         }
143 
144     }
