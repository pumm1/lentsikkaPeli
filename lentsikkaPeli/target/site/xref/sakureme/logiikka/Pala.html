1   package sakureme.logiikka;
2   
3   import java.awt.Graphics;
4   
5   /*
6   Tässä luodaan pelin palojen (ja ns. 'hitboxien') logiikka
7    */
8   public class Pala {
9   
10      private int x;
11      private int y;
12      private int leveys;
13      private int korkeus;
14      private int dy = 0;     //putoamista varten, kaikki palat eivät välttämättä liiku
15  
16      public Pala(int ax, int yy, int w, int h) {
17          x = ax;
18          y = yy;
19          leveys = w;
20          korkeus = h;
21      }
22  
23      ////////////getterit///////////////
24      public int getX() {
25          return x;
26      }
27  
28      public int getY() {
29          return y;
30      }
31  
32      public int getLeveys() {
33          return leveys;
34      }
35  
36      public int getKorkeus() {
37          return korkeus;
38      }
39  
40      ////kahden palasen osumisen logiikka////
41      public boolean osuuko(Pala i) {
42  
43          if (x + leveys >= i.getX() && x <= i.getX() + i.getLeveys()) {
44              if (y + korkeus >= i.getY() && y <= i.getY() + i.getKorkeus()) {
45                  return true;
46              }
47          }
48  
49          return false;
50      }
51  
52      /////palan liikukuminen////////////
53      public void liikuX(int i) {
54          x += i;
55      }
56  
57      public void liikuY(int j) {
58          y += j;
59      }
60  
61      //////palan piirtäminen///////
62      //alustavasti hyödynnetään Javan Graphics-oliota pelin testaamiseen
63      //pyritään käyttämään lopussa .png-kuvia
64      //pala koostuu neliöstä, jonka vasen ylänurkka koordinaateissa (x,y) ja koko leveys*korkeus (voidaan tarvittaessa muuttaa)
65      public void piirra(Graphics g) {
66          g.fillRect(x, y, leveys, korkeus);
67  
68      }
69  
70  }
